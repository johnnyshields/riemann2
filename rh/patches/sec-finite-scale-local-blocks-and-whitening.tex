\label{sec:finite-scale-local-blocks-and-whitening}

\par\medskip

Around a midpoint \(m\) at finite separation \(s>0\), the same-point
and cross-block jets of Section~\ref{sec:jet-limit-local-blocks}
extend to two same-point Gram blocks \(G_{m,\pm}(s)\) and a mixed
block \(N_m(s)\); whitening these into the dimensionless
\(\widehat\Om_\z(s;m)\)
(Definition~\ref{def:whitened-finite-block}) with the
polynomial-in-\(Q\) whitening-loss bound of
Lemma~\ref{lem:whitening-loss} produces the input consumed by every
comparison theorem from
Section~\ref{sec:toy-anomaly-and-transverse-obstruction} onward.

\subsection{Finite same-point and mixed blocks}
\label{subsec:finite-same-point-and-mixed-blocks}

Fix a midpoint \(m\) and a separation \(s>0\); set \(t_\pm = m\pm s/2\),
\(q_\pm = q(t_\pm)\), and \(\Del_m(s) = \Ph(t_-) - \Ph(t_+)\).

\begin{definition}[Finite same-point and mixed blocks]
\label{def:finite-same-point-and-mixed-blocks}
\begin{equation}
\label{eq:Gpm}
G_{m,\pm}(s)
=\frac1\pi
\begin{pmatrix}
2q(t_\pm) & \tfrac12 q'(t_\pm)\\[0.8ex]
\tfrac12 q'(t_\pm) & \tfrac1{12}\bigl(q''(t_\pm)+2q(t_\pm)^3\bigr)
\end{pmatrix},
\end{equation}
\begin{equation}
\label{eq:Nm}
N_m(s)
=\frac1\pi
\begin{pmatrix}
-\dfrac{2\sin\Del_m(s)}{s}
&
\dfrac{\sin\Del_m(s)+q_+ s\cos\Del_m(s)}{s^2}
\\[2.4ex]
-\dfrac{\sin\Del_m(s)+q_- s\cos\Del_m(s)}{s^2}
&
\dfrac{(q_-+q_+)s\cos\Del_m(s)+(2-q_- q_+ s^2)\sin\Del_m(s)}{2s^3}
\end{pmatrix}.
\end{equation}
\end{definition}

\begin{remark}[Connection to \S\ref{sec:jet-limit-local-blocks}]
\label{rem:finite-blocks-from-jet-limit}
The blocks of Definition~\ref{def:finite-same-point-and-mixed-blocks}
are the kernel-derivative form of the same-point and cross-block jets
of \S\ref{sec:jet-limit-local-blocks} evaluated at the centers
\(t_\pm = m\pm s/2\).  Concretely, \(G_{m,\pm}(s) = J(t_\pm)\) is the
same-point Gram block of \eqref{eq:same-point-J} at \(T = t_\pm\), and
\(N_m(s) = \pi^{-1} N_{12}(t_-, t_+)\) is the cross-block of
\eqref{eq:cross-N12} at \((T_1, T_2) = (t_-, t_+)\), with
\(s_{12} = t_- - t_+ = -s\) absorbed into the displayed signs.
\end{remark}

\begin{lemma}[Removable singularity of \(N_m(s)\)]
\label{lem:Nm-removable-singularity}
If \(\Ph\in C^4\) on a neighborhood of \(m\), the entries of
\(N_m(s)\) of \eqref{eq:Nm} extend continuously to \(s=0\) with
\begin{equation}
\label{eq:Nm-at-zero}
        N_m(0) = J(m)
        = \frac1\pi
        \begin{pmatrix}
        2q(m) & \tfrac12 q'(m)\\[0.8ex]
        \tfrac12 q'(m) & \tfrac1{12}\bigl(q''(m)+2q(m)^3\bigr)
        \end{pmatrix}.
\end{equation}
\end{lemma}

\begin{proof}
Taylor expansion of \(\Ph\) at \(m\) gives
\(\Del_m(s) = -s q(m) - (s^3/24) q''(m) + O(s^5)\) and \(q_\pm = q(m)\mp
(s/2)q'(m) + O(s^2)\).  Substituting into \eqref{eq:Nm} and using
\(\sin z = z - z^3/6 + O(z^5)\) cancels the displayed denominators.
The four resulting limits are
\[
        \lim_{s\to 0}\!\Bigl(-\tfrac{2\sin\Del_m(s)}{\pi s}\Bigr) = \tfrac{2q(m)}{\pi},
\]
\[
        \lim_{s\to 0}\!\tfrac{\sin\Del_m(s)+q_+ s\cos\Del_m(s)}{\pi s^2} = \tfrac{q'(m)}{2\pi},
\]
\[
        \lim_{s\to 0}\!\Bigl(-\tfrac{\sin\Del_m(s)+q_- s\cos\Del_m(s)}{\pi s^2}\Bigr) = \tfrac{q'(m)}{2\pi},
\]
\[
        \lim_{s\to 0}\!\tfrac{(q_-+q_+)s\cos\Del_m(s)+(2-q_-q_+s^2)\sin\Del_m(s)}{2\pi s^3}
        = \tfrac{q''(m)+2q(m)^3}{12\pi},
\]
matching \(J(m)\) entry by entry.
\end{proof}

\subsection{Whitened finite block}
\label{subsec:whitened-finite-block}

\begin{definition}[Whitened finite block]
\label{def:whitened-finite-block}
Provided \(G_{m,\pm}(s)\) are positive definite, set
\begin{equation}
\label{eq:omega-finite}
\widehat\Om_\z(s;m)
:= G_{m,-}(s)^{-1/2}\,N_m(s)\,G_{m,+}(s)^{-1/2}.
\end{equation}
\end{definition}

\begin{corollary}[Coalescence: \(\widehat\Om_\z(0;m) = I_2\)]
\label{cor:omega-coalescence}
For every retained midpoint \(m\) at sufficiently large height,
\(\widehat\Om_\z(s;m)\) extends continuously to \(s=0\) with
\(\widehat\Om_\z(0;m) = I_2\).
\end{corollary}

\begin{proof}
At \(s=0\), \(t_\pm \to m\) gives \(G_{m,\pm}(0) = J(m)\), and
Lemma~\ref{lem:Nm-removable-singularity} gives \(N_m(0) = J(m)\).
Substituting into \eqref{eq:omega-finite},
\(\widehat\Om_\z(0;m) = J(m)^{-1/2}\,J(m)\,J(m)^{-1/2} = I_2\), where
the inverse exists by Lemma~\ref{lem:same-point-gram-positivity}.
\end{proof}

\subsection{Whitening loss}
\label{subsec:whitening-loss}

\begin{lemma}[Operator-norm bound for \(G_{m,\pm}^{-1/2}\)]
\label{lem:whitening-loss}
For \(\Ph_T = \theta\) on \(I_T\) and any retained midpoint \(m\) with
\(t_\pm = m\pm s/2\in I_T\), the operator norm of the whitening factor
satisfies
\begin{equation}
\label{eq:whitening-loss-bound}
        \bigl\|G_{m,\pm}(s)^{-1/2}\bigr\|_{\op}
        \;\le\;
        \sqrt{\tfrac{\pi}{2\,q(t_\pm)}}\,(1 + o(1))
        \qquad (T\to\infty),
\end{equation}
uniformly in \((m,s)\) on the retained set.  Above the global
lower-height cutoff of
Lemma~\ref{lem:same-point-gram-positivity}, this gives the
polynomial-in-\(Q\) bound
\begin{equation}
\label{eq:whitening-loss-polynomial}
        \bigl\|G_{m,\pm}(s)^{-1/2}\bigr\|_{\op}
        \;\le\; Q(T)^{C_W}
\end{equation}
with absolute exponent \(C_W = 0\) whenever \(Q(T)\ge 1\).
\end{lemma}

\begin{proof}
By Remark~\ref{rem:finite-blocks-from-jet-limit},
\(G_{m,\pm}(s) = J(t_\pm)\); applying
Lemma~\ref{lem:same-point-gram-positivity} at \(T = t_\pm\) gives
\(\lambda_{\min}(G_{m,\pm}(s)) = (2q(t_\pm)/\pi)(1 + o(1))\) at large
heights.  For a positive-definite symmetric \(2\times 2\) matrix \(G\),
\(\|G^{-1/2}\|_{\op} = \lambda_{\min}(G)^{-1/2}\), whence
\eqref{eq:whitening-loss-bound}.  Above the cutoff of
Lemma~\ref{lem:same-point-gram-positivity},
\(\lambda_{\min}(G_{m,\pm})\ge 1\), so
\(\|G_{m,\pm}^{-1/2}\|_{\op}\le 1\le Q(T)^0\), giving \(C_W = 0\).
\end{proof}

\begin{remark}[Scope of Lemma~\ref{lem:whitening-loss}]
\label{rem:whitening-loss-scope}
The bound \eqref{eq:whitening-loss-bound} consumes only
\eqref{eq:lambda-min-asymptotic} of
Lemma~\ref{lem:same-point-gram-positivity}, that is, the theta-phase
Gram positivity proven in \S\ref{sec:jet-limit-local-blocks}.  No
omitted-zero, no zero-free-region input, and no holomorphic-extension
machinery is consumed here.  The corrected-whitening machinery of v1,
which uses an omitted-zero shell decomposition to produce sharper
constants on a complex disk \(|s|<\rho_0/Q\), is a separate piece of
analysis whose migration is independent of the present whitening
bound.
\end{remark}
