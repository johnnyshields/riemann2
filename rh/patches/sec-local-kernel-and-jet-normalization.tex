\section{Local kernel and jet normalization}
\label{sec:local-kernel-and-jet-normalization}
\statuspaper{LIVE}\quad
\statusmig{migrated}\quad
\statusreferee{not-started}\quad
\statussympy{verified}\quad

\par\medskip

Fix a height \(\gamma\) and let \(T = \gamma\).  We work on a window
\(I_T \subset \mathbb R\) centered at \(T\) of length
\(\asymp 1/Q(T)\), where \(Q\) is a fixed slowly growing function of
the height.  All subsequent local estimates are expressed through the
phase kernel \(K_\Ph\) on \(I_T\times I_T\) and its Taylor jets at
one or two centers.

\subsection{Local phase chart}
\label{subsec:local-phase-chart}

\begin{definition}[Local phase chart]
\label{def:local-phase-chart}
For each height \(T\), \(\Ph_T\) is a real smooth function on an
interval \(I_T\) of length \(\asymp 1/Q(T)\) centered at \(T\)
satisfying:
\begin{enumerate}[label=\textup{(P\arabic*)},leftmargin=*]
\item \(\Ph_T = \theta\), the Riemann--Siegel theta function;
\item \(q := \Ph_T'\) admits a polynomial-in-\(Q\) lower bound on
  retained packets, \(q(t) \ge Q^{-c_q}\) for an absolute exponent
  \(c_q \ge 0\) (Lemma~\ref{lem:phase-derivative-lower-bound}).
\end{enumerate}
We write \(\Ph,q\) for \(\Ph_T,q\) when \(T\) is fixed by context.
\end{definition}

\begin{remark}[Identification \(\Ph_T = \theta\)]
\label{rem:phase-chart-identification}
The functional equation of \(\zeta\) gives
\[
\zeta(\tfrac12 + it) = e^{-i\theta(t)}\, Z(t),
\qquad
\theta(t) = \arg\Gamma\!\left(\tfrac14 + \tfrac{it}{2}\right) - \frac{t}{2}\log\pi,
\]
with \(Z\) real on \(\mathbb R\); we set \(\Ph_T(t) := \theta(t)\) on
\(I_T\).
\end{remark}

\begin{lemma}[Phase-derivative lower bound]
\label{lem:phase-derivative-lower-bound}
With \(\Ph_T = \theta\) and \(Q\) polynomial in \(\log T\),
\(q(t) \ge Q^{-c_q}\) holds with \(c_q = 0\) on \(I_T\) for all
sufficiently large \(T\).  Concretely,
\[
q(t) \ge \tfrac12 \log(T/(4\pi)) - C\,T^{-2}
\qquad (t \in I_T)
\]
for an absolute constant \(C\).
\end{lemma}

\begin{proof}
The Riemann--Siegel asymptotic
\cite{Edwards2001,Titchmarsh1986}
\[
\theta(t) = \frac{t}{2}\log\frac{t}{2\pi} - \frac{t}{2} - \frac{\pi}{8}
          + \frac{1}{48 t} + O(t^{-3})
\]
gives \(q(t) = \theta'(t) = \tfrac12\log(t/(2\pi)) - 1/(48 t^2) +
O(t^{-4})\).  On \(I_T\), \(t \ge T - 1/Q \ge T/2\) once
\(Q \ge 2/T\), and the displayed bound follows.
\end{proof}

\begin{lemma}[Phase-derivative upper bounds]
\label{lem:phase-derivative-upper-bounds}
With \(\Ph_T = \theta\),
\[
q'(T) \;=\; \frac{1}{2T} + O(T^{-3}),\qquad
q''(T) \;=\; -\frac{1}{2T^2} + O(T^{-4}).
\]
\end{lemma}

\begin{proof}
Termwise differentiation of the Riemann--Siegel asymptotic in the
proof of Lemma~\ref{lem:phase-derivative-lower-bound}.
\end{proof}

\begin{remark}[\(\zeta\)-side regularity]
\label{rem:chart-regularity-zeta-side}
The chart depends on \(\zeta\) only through
\(\zeta(\tfrac12 + it) = e^{-i\theta(t)} Z(t)\).  Quantitative
\(\zeta\)-side conditions (lower bounds on \(|Z|\), \(|\zeta|\), or
no-zeros conditions on \(I_T\)) are imposed where they are consumed
in
Section~\ref{sec:actual-zeta-transverse-suppression} and
Section~\ref{sec:source-energy-supply}, not at chart construction.
\end{remark}

\begin{definition}[Phase kernel]
\label{def:phase-kernel}
\begin{equation}
\label{eq:phase-kernel}
K_\Ph(x,y)=
\begin{cases}
\dfrac{\sin\bigl(\Ph(x)-\Ph(y)\bigr)}{\pi(x-y)}, & x\ne y,\\[1.2ex]
\dfrac{q(x)}{\pi}, & x=y.
\end{cases}
\end{equation}
\end{definition}

\begin{lemma}[Phase-kernel basic properties]
\label{lem:phase-kernel-properties}
For \(\Ph\in C^3(I_T;\mathbb R)\), the kernel \(K_\Ph\) of
\eqref{eq:phase-kernel} satisfies:
\begin{enumerate}[label=\textup{(\roman*)},leftmargin=*]
\item \(K_\Ph(x,y) = K_\Ph(y,x)\);
\item \(K_\Ph(x,y)\in\mathbb R\) for \((x,y)\in I_T\times I_T\);
\item \(\lim_{x\to y} K_\Ph(x,y) = q(y)/\pi\), continuously across
  the diagonal.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) For \(x\ne y\),
\(\sin(\Ph(y)-\Ph(x))/\pi(y-x) = \sin(\Ph(x)-\Ph(y))/\pi(x-y)\).
(ii) The numerator and denominator are real.
(iii) Taylor expand: \(\Ph(y+h)-\Ph(y) = h q(y) + O(h^2)\), so
\(K_\Ph(y+h,y) = q(y)/\pi + O(h)\).
\end{proof}

\begin{lemma}[Phase-kernel derivatives]
\label{lem:phase-kernel-derivatives}
For \(T_1\ne T_2\) with \(s = T_1 - T_2\),
\(\Del = \Ph(T_1) - \Ph(T_2)\), and \(q_j = q(T_j)\),
\begin{align*}
K_\Ph(T_1, T_2) &= \frac{\sin\Del}{\pi s},
&K_x(T_1, T_2) &= \frac{q_1\, s\cos\Del - \sin\Del}{\pi s^2},\\
K_y(T_1, T_2) &= \frac{\sin\Del - q_2\, s\cos\Del}{\pi s^2},
&K_{xy}(T_1, T_2)
&= \frac{(q_1+q_2)s\cos\Del + (q_1 q_2 s^2 - 2)\sin\Del}{\pi s^3}.
\end{align*}
\end{lemma}

\begin{proof}
Direct application of the quotient and chain rules to
\(K_\Ph(x,y) = \sin(\Ph(x)-\Ph(y))/\pi(x-y)\).
\end{proof}

\begin{remark}[Diagonal limits]
\label{rem:kernel-derivatives-diagonal}
As \(T_2\to T_1 = T\) the formulas of
Lemma~\ref{lem:phase-kernel-derivatives} have limits
\(K_\Ph(T,T) = q(T)/\pi\),
\(K_x(T,T) = K_y(T,T) = q'(T)/(2\pi)\),
\(K_{xy}(T,T) = (q''(T) + 2q(T)^3)/(6\pi)\),
giving the kernel-derivative form of \(J(T)\) used in
\eqref{eq:same-point-as-kernel-derivatives}.
\end{remark}

\subsection{Point-to-jet transform}
\label{subsec:point-to-jet-transform}

For symmetric pairs \(T\pm h\), define the Gram-normalized
point-to-jet transform
\begin{equation}
\label{eq:point-to-jet-transform}
P_h
\;=\;
\frac1{\sqrt2}
\begin{pmatrix}
1 & 1\\[0.6ex]
-1/(2h) & 1/(2h)
\end{pmatrix}
\;=\;
\diag(\sqrt 2,\, 1/\sqrt 2)\,
\begin{pmatrix}
1/2 & 1/2\\[0.6ex]
-1/(2h) & 1/(2h)
\end{pmatrix},
\end{equation}
the second equality exhibiting \eqref{eq:point-to-jet-transform} as
the diagonal congruence of the literal value/derivative jet
transform.  With this convention,
\begin{equation}
\label{eq:same-point-as-kernel-derivatives}
J(T)
\;=\;
\begin{pmatrix}
2K_\Ph(T,T) & \partial_y K_\Ph(T,T)\\[0.6ex]
\partial_x K_\Ph(T,T) & \tfrac12 \partial_{xy} K_\Ph(T,T)
\end{pmatrix},
\end{equation}
which agrees with the explicit formula
\eqref{eq:same-point-J} of
Lemma~\ref{lem:same-point-jet-limit}.

\begin{remark}[Pair separation]
\label{rem:pair-separation-convention}
We use \(T\pm h\), so the pair separation is \(2h\) and
\(P_h\) carries \(1/(2h)\) (not \(1/h\)) in its lower row.  Imported
results stated in the alternative \(T\pm h/2\) convention are to be
retransformed before use.
\end{remark}

\begin{remark}[Convention check]
\label{rem:normalization-audit}
The literal-jet alternative
\(\tfrac12\bigl(\begin{smallmatrix}1&1\\-1/h&1/h\end{smallmatrix}\bigr)\)
applied to \(A_h(T)\) at the same pair \(T\pm h\) yields the limit
\[
\frac1\pi
\begin{pmatrix}
q(T) & \tfrac12 q'(T)\\[0.6ex]
\tfrac12 q'(T) & \tfrac16\bigl(q''(T)+2q(T)^3\bigr)
\end{pmatrix},
\]
related to \(J(T)\) of \eqref{eq:same-point-J} by the diagonal
congruence \(\diag(\sqrt 2, 1/\sqrt 2)\) of
\eqref{eq:point-to-jet-transform}.  The Gram-normalized form is used
throughout this paper.
\end{remark}
